# session_exporter/validators/chat_parser.py

def parse_transcript(text):
    turns = []
    lines = text.splitlines()
    role = None
    buffer = []

    for line in lines:
        line_strip = line.strip()
        if line_strip.startswith("You said:"):
            if buffer: turns.append((role, "\n".join(buffer)))
            role = "User"
            buffer = []
        elif line_strip.startswith("ChatGPT said:"):
            if buffer: turns.append((role, "\n".join(buffer)))
            role = "Assistant"
            buffer = []
        else:
            buffer.append(line)

    if buffer: turns.append((role, "\n".join(buffer)))
    return turns

import re

def clean_ui_noise(text):
    noisy_lines = [
        "Open sidebar", "Search", "Create image", "ChatGPT can make mistakes",
        "Share", "Skip to content", "Deep research"
    ]
    return "\n".join(
        line for line in text.splitlines()
        if not any(noise in line for noise in noisy_lines)
    )

# session_exporter/validators/fence_validator.py

def validate_fences(parsed_turns):
    for role, content in parsed_turns:
        fence_errors = [
            line for line in content.splitlines()
            if line.count("```") == 1 or line.strip().startswith("`{")
        ]
        if fence_errors:
            raise ValueError(f"Invalid fence in {role} turn:\n" + "\n".join(fence_errors))

# session_exporter/exporters/markdown_writer.py

from jinja2 import Template
from pathlib import Path

def write_markdown(turns, output_path):
    nav = "\n".join([f"- [Turn {i+1} - {r}](#turn-{i+1}---{r.lower()})" for i, (r, _) in enumerate(turns)])
    body = ""
    for i, (role, content) in enumerate(turns):
        header = f"## Turn {i+1} - {role}"
        body += f"\n\n{header}\n\n```text\n{content.strip()}\n```"

    template_path = Path(__file__).parent.parent / "templates/session_template.md.j2"
    with open(template_path) as f:
        template = Template(f.read())

    result = template.render(
        version="v0.1.0",
        navigation=nav,
        body=body
    )

    Path(output_path).write_text(result, encoding="utf-8")
